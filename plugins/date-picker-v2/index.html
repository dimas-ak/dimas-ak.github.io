Handlers

Format Date

type : STRING 
// Default : "mm-dd-yyyy"
// Another formatDate :
"mm-dd-yyyy", // bulan-hari-tahun
"mm-yyyy-dd", // bulan-tahun-hari
"dd-mm-yyyy", // hari-bulan-tahun
"dd-yyyy-mm", // hari-tahun-bulan
"yyyy-mm-dd", // tahun-bulan-hari
"yyyy-dd-mm", // tahun-hari-bulan

"mm.dd.yyyy", // bulan.hari.tahun
"mm.yyyy.dd", // bulan.tahun.hari
"dd.mm.yyyy", // hari.bulan.tahun
"dd.yyyy.mm", // hari.tahun.bulan
"yyyy.mm.dd", // tahun.bulan.hari
"yyyy.dd.mm", // tahun.hari.bulan


"mm/dd/yyyy", // bulan/hari/tahun
"mm/yyyy/dd", // bulan/tahun/hari
"dd/mm/yyyy", // hari/bulan/tahun
"dd/yyyy/mm", // hari/tahun/bulan
"yyyy/mm/dd", // tahun/bulan/hari
"yyyy/dd/mm", // tahun/hari/bulan

// Example :
formatDate : "yyyy-mm-dd"

Range Years

type : STRING 
// Default : Last Year and Now

// Particular Years and ':' for Separator
rangeYears : "1980:2005"

// OR choose your last year
rangeYears :"1980" // It mean that 1980 until Now

Show Date

type : ARRAY => BOOLEAN, BOOLEAN 
// Default : [ true, false ]
// First Boolean : Date Picker
// Second Boolean : Time Chooser

showDate : 
[
    true, // => Set it to false if you don't want to see the Date
    true // => Set it to true if you want to see the Time
]

// Example :  if you want to show the Time Chooser only, then
showDate :
[
    false,
    true
]

Output:

showDate : 
[ 
    false,
    true
]

Minutes & Hours

type : STRING OR ARRAY => INT 
// Default hours    : "0-24" <= It mean 0-23
// Default minutes  : "0-60" <= It mean 0-59

//Let's say, you want to set certain minutes OR hours

minutes    : [0, 15, 30, 45, 60],
hours      : [8, 9, 10, 11, 12]

/*
NOTE
MAX of Hours is 23
MAX of Minutes is 59
*/

Holidays

type : ARRAY => STRING, STRING 
// I'll update it for every year
// ["month-day-year", "the name of your holiday"]
// Default holidays :
["00-25-2020", "Hari Raya Imlek"],
["02-22-2020", "Isra Mi'raj"],
["02-25-2020", "Hari Raya Nyepi"],
["03-10-2020", "Paskah (Jumat Agung)"],
["04-01-2020", "Hari Buruh Nasional"],
["04-07-2020", "Hari Raya Waisak"],
["04-21-2020", "Kenaikan Yesus"],
["04-22-2020", "Cuti Bersama Lebaran"],
["04-24-2020_04-25-2020", "Hari Raya Idul Fitri"],
["04-26-2020_04-27-2020", "Cuti Bersama Lebaran"],
["06-31-2020", "Idul Adha"],
["07-20-2020", "Tahun Baru Islam"],
["09-29-2020", "Maulid Nabi Muhammad SAW"],
["11-24-2020", "Cuti Bersama Hari Raya Natal"],

// NOTE: First value of month is zero ("00" OR January) until 11 (December)

/*
    Let's say, you want to put some holidays in sequence
    Separator: "_" OR underline
    ["month-day-year_month-day-year", "the name of your holidays"]
*/

// Example
holidays :
[
    ["01-15-2020_01-20-2020", "My Holidays"] // It mean that date from 15 - 20 February 2020
]

// Set to FALSE if you don't want to show the Holidays
holidays : false
